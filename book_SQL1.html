SQL基本1

SELECT


SELECT *
SELECT name, price

WHERE


-- 食費を探す
WHERE category = "食費";

-- 1000以上のデータを検索する
WHERE price >= 1000;

--"2017-07-01"を検索
WHERE day = "2017-07-01";

-- 2017-07-01以前のデータを検索
WHERE day <= "2017-07-01" ;

LIKE


-- 前後のワイルドカードでプリンを含むデータを全て取得する
WHERE name LIKE "%プリン%";

-- プリンで始まるデータを取得する #プリン、プリンパフェ"
WHERE name LIKE "プリン%";

-- プリンで終わるデータを取得する #プリン、牛乳プリン
WHERE name LIKE "%プリン";

NOT


-- 1000以上のデータを満たさないデータを取得する
WHERE NOT price > 1000;

-- 条件を満たさないデータを取得できる
WHERE NOT name LIKE "%プリン%"

NULL


-- 指定したカラムのデータがNULLである物を取得
WHERE price IS NULL;

-- 指定したカラムのデータがNULLではない物を取得する
WHERE price IS NOT NULL

-- NULLの物を取得
WHERE price = NULL;

-- NULLではないものを取得する
WHERE NOT price = NULL;

AND, OR


-- #二つが一致するデータを検索
WHERE name = "hoge"
AND category = "食費";

-- 何方かが一致するデータを検索
WHERE name = "hoge"
OR category = "食費";

ORDER BY


ASC...昇順 1,2,3,4
